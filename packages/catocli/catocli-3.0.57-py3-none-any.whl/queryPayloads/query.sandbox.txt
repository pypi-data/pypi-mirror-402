query sandbox ( $accountId:ID! $sandboxReportsInput:SandboxReportsInput!) {
	sandbox ( accountId:$accountId ) {
		reports ( input:$sandboxReportsInput  )  {
			report {
				fileName
				fileHash
				creationDate
				expirationDate
				verdict
				status
				downloadUrl
				failureReason
			}
			pageInfo {
				total
			}
		}
	}	
}
