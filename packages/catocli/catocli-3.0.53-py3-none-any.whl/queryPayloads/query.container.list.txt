query containerList ( $accountId:ID! $containerSearchInput:ContainerSearchInput!) {
	container ( accountId:$accountId ) {
		list ( input:$containerSearchInput  )  {
			containers {
				id
				name
				description
				size
				audit {
					createdBy
					lastModifiedBy
					lastModifiedAt
					createdAt
				}
				syncData {
					fileType
					url
					timeInterval
					timeUnit
					notifications {
						subscriptionGroup {
							id
							name
						}
						webhook {
							id
							name
						}
						mailingList {
							id
							name
						}
					}
				}
				syncDataAudit {
					lastSynced
					lastSyncAttempt
					errorMsg
				}
			}
		}
	}	
}
