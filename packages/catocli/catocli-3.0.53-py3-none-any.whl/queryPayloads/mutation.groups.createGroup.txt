mutation groupsCreateGroup ( $accountId:ID! $createGroupInput:CreateGroupInput! $groupMembersListInput:GroupMembersListInput!) {
	groups ( accountId:$accountId ) {
		createGroup ( input:$createGroupInput  )  {
			group {
				id
				name
				description
				audit {
					updatedTime
					updatedBy {
						id
						name
					}
				}
				members ( input:$groupMembersListInput  )  {
					items {
						id
						name
						type
					}
					paging {
						total
					}
				}
				membersCount
				membersCountPerType {
					type
					membersCount
				}
			}
		}
	}	
}
