query popLocations ( $accountId:ID! $popLocationFilterInput:PopLocationFilterInput) {
	popLocations ( accountId:$accountId ) {
		popLocationList ( input:$popLocationFilterInput  )  {
			items {
				id
				name
				displayName
				country {
					id
					name
				}
				isPrivate
				cloudInterconnect {
					taggingMethod
					providerName
				}
			}
		}
	}	
}
