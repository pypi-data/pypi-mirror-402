query groupsGroupMembers ( $accountId:ID! $groupRefInput:GroupRefInput! $groupMembersListInput:GroupMembersListInput!) {
	groups ( accountId:$accountId ) {
		group ( input:$groupRefInput  )  {
			members ( input:$groupMembersListInput   ) {
				items {
					id
					name
					type
				}
				paging {
					total
				}
			}
		}
	}	
}
